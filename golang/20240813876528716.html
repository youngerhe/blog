	for {

		case <-ticker.C:

			h.mu.RLock()

				delete(h.clients, client)

				// å¦‚æœæˆ¿é—´è¿˜æœ‰äººé€šçŸ¥ä»–ä¸‹çº¿

func (c *Client) Heartbeat() {

	defer func() {
		ticker.Stop()

			log.Printf("Heartbeat recover: %v", r)
			if c.Conn != nil {
				// TODO è®°å½•æ—¥å¿—

				if err != nil {
					log.Printf("Heartbeat Error closing connection: %v", err)

		case <-ticker.C:

					if c.RoomId != "" {
						MyServer.Hubs[c.RoomId].unregister <- c
					}
					return
				}
			} else {
				// å¦‚æœæˆåŠŸåˆ™é‡ç½®ä¸º0
				failTimes = 0
				//log.Printf("å‘é€å¿ƒè·³æˆåŠŸ,ç”¨æˆ·idï¼š%v", c.UserId)
			}
		}
	}
