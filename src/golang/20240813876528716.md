```golang
	MyServer = NewServer()
	return MyServer
}

	return &Server{
		Hubs: make(map[string]*Hub),
	}
}

	hub := NewHub()
	s.rwLock.RLock()
	s.Hubs["register"] = hub
	s.rwLock.RUnlock()
```

// hub.go，维护不同的房间，可以生成多个通话频道

```golang
package ws

import (
	"call/internal/types"
	"encoding/json"
	"fmt"
	"log"
	"sync"
	"time"
)

type Hub struct {
	// 注册到Hub的客户端
	clients map[*Client]bool

	// 用于在多个goroutine间同步
	mu sync.RWMutex

	// 广播消息到所有连接的客户端
	broadcast chan []byte

	// 注册和注销的通道
	register   chan *Client
	unregister chan *Client
	master     string
	viewer     string
	id         string
	closeCh    chan struct{}
}

func NewHub() *Hub {
	return &Hub{
		clients:    make(map[*Client]bool),
		broadcast:  make(chan []byte),
		register:   make(chan *Client),
		unregister: make(chan *Client),
		closeCh:    make(chan struct{}),
	}
}

// Run 启动这个hub的注册退出程序
func (h *Hub) Run() {
	ticker := time.NewTicker(2 * time.Second)
	for {
		select {
		case <-ticker.C:

			log.Print("协程", h.id)
			// 执行不到这就被关闭了，不写还不行。。
		case <-h.closeCh:
			h.clean()
			log.Print("关闭协程", h.id)
			return
		case client := <-h.register:

			h.mu.RLock()
			h.clients[client] = true

			msg := &types.JoinResp{
				Event: "join",
				Data: types.JoinRespData{
					Count:  len(h.clients),
					Master: h.master,
					Viewer: h.viewer,
				},
			}
			msgByte, err := json.Marshal(msg)
			if err != nil {
				// TODO 记录日志
				continue
			}

			for c := range h.clients {
				if client != c {
					c.Send <- msgByte
				}
			}
			h.mu.RUnlock()

		case client := <-h.unregister:
			h.mu.RLock()
			if _, ok := h.clients[client]; ok {
				client.RoomId = ""
				delete(h.clients, client)

				// 如果房间还有人通知他下线
				if len(h.clients) > 0 {
					msg := &types.JoinResp{
						Event: "exit_room",
					}
					msgByte, _ := json.Marshal(msg)
					for c := range h.clients {
						c.Send <- msgByte
					}

				} else {
					//无人，关闭这个房间
					MyServer.Hubs[h.id].Close()
				}

			}
			h.mu.RUnlock()

		case message := <-h.broadcast:
			fmt.Println("广播消息")
			h.mu.RLock()
			for client := range h.clients {
				select {
				case client.Send <- message:
				default:
					delete(h.clients, client)
				}
			}
			h.mu.RUnlock()
		}
	}
}

func (h *Hub) SendJoinMsg() {

	msg := &types.JoinResp{
		Event: "join",
		Data: types.JoinRespData{
			Count:  len(h.clients),
			Master: h.master,
			Viewer: h.viewer,
		},
	}
	msgByte, err := json.Marshal(msg)
	if err != nil {
		// TODO 记录日志
		return
	}
	h.mu.RLock()
	h.broadcast <- msgByte
	h.mu.RUnlock()

}

func (h *Hub) Close() {

	//关闭资源
	if _, ok := MyServer.Hubs[h.id]; ok {
		delete(MyServer.Hubs, h.id)
		h.closeCh <- struct{}{}
	}
}

func (h *Hub) clean() {
	close(h.broadcast)
	close(h.register)
	close(h.unregister)
}
```

```golang
package ws

import (
	"call/internal/types"
	"encoding/json"
	"fmt"
	"github.com/gin-gonic/gin"
	"github.com/gorilla/websocket"
	"log"
	"net/http"
	"time"
)

type Client struct {

	// WebSocket连接
	Conn *websocket.Conn

	// 发送消息给客户端的通道
	Send chan []byte

	// 设备id / 用户id
	UserId string
	RoomId string
}

func NewClient(c *websocket.Conn) *Client {

	return &Client{
		Conn: c,
		Send: make(chan []byte),
	}
}

// ClientHandle 连接处理
func ClientHandle(c *gin.Context) {

	// 升级协议
	var upgrader = websocket.Upgrader{CheckOrigin: func(r *http.Request) bool {
		return true
	}}
	conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
	if err != nil {
		return
	}
	client := NewClient(conn)

	// 注册到注册频道, 管理全部用户
	// MyServer.Hubs["register"].register <- client

	// 为每一个client 处理读写
	go client.ReadPump()
	go client.WritePump()
	go client.Heartbeat()

}

func (c *Client) ReadPump() {
	defer func() {
		if r := recover(); r != nil {

			// TODO 记录日志
			err := c.Conn.Close()
			if err != nil {
				log.Printf("ReadPump Error closing connection after panic: %v", err)
				return
			}

		}
	}()
	for {
		_, message, err := c.Conn.ReadMessage()
		if err != nil {
			log.Printf("c.Conn.ReadMessage() 触发错误，%v:", err)
			err = c.Conn.Close()
			if err != nil {
				log.Printf("SetReadDeadline c.Conn.Close(): %v", err)
				return
			}
		}
		var req = types.WsReq{}
		err = json.Unmarshal(message, &req)
		if err != nil {
			// TODO 打印日志
			continue
		}
		switch req.Event {
		case "join":
			log.Printf("加入房间：%v", c.UserId)
			err = c.joinHandle(message)
			if err != nil {
				// TODO 打印日志
				fmt.Println(err)
			}

		case "offer_sdp":
			var signallingReq = types.SignallingReq{}
			err = json.Unmarshal(message, &signallingReq)
			if err != nil {
				// TODO 打印日志
				fmt.Println(err)
				continue
			}
			hub := MyServer.Hubs[signallingReq.Data.RoomId]
			for client := range hub.clients {
				if client != c {
					client.Send <- message
				}
			}

		case "answer_sdp":
			var signallingReq = types.SignallingReq{}
			err = json.Unmarshal(message, &signallingReq)
			if err != nil {
				// TODO 打印日志
				fmt.Println(err)

			}
			hub := MyServer.Hubs[signallingReq.Data.RoomId]
			for client := range hub.clients {
				if client != c {
					client.Send <- message
				}
			}

		case "candidate":
			var signallingReq = types.SignallingReq{}
			err = json.Unmarshal(message, &signallingReq)
			if err != nil {
				// TODO 打印日志
				fmt.Println(err)
				continue
			}
			hub := MyServer.Hubs[signallingReq.Data.RoomId]
			for client := range hub.clients {
				if client != c {
					client.Send <- message
				}
			}

		case "exit":
			// 用户退出房间
			if _, ok := MyServer.Hubs[c.RoomId]; ok {

				hub := MyServer.Hubs[c.RoomId]
				if _, ok := hub.clients[c]; ok {
					hub.unregister <- c
				}
			}

		}
	}
}

// 加入房间处理
func (c *Client) joinHandle(data []byte) error {

	var req types.JoinReq
	err := json.Unmarshal(data, &req)
	if err != nil {
		fmt.Println("err")
		return err
	}
	var hub *Hub
	MyServer.rwLock.RLock()
	if myServerHub, ok := MyServer.Hubs[req.Data.RoomId]; !ok {
		hub = NewHub()
		MyServer.Hubs[req.Data.RoomId] = hub
		hub.id = req.Data.RoomId
		go hub.Run()
	} else {
		hub = myServerHub
	}
	MyServer.rwLock.RUnlock()
	c.UserId = req.Data.UserId
	// 已经加入房间的用户 不能加入其他房间，不处理就行了，因为正常逻辑的用户不能重复加入
	if c.RoomId != "" {
		return nil
	}
	c.RoomId = req.Data.RoomId

	if hub.viewer == c.UserId || hub.master == c.UserId {
		return nil
	}
	if len(hub.clients) == 2 {
		return nil
	}
	if len(hub.clients) == 1 {

		//  TODO 判断该游客权限，是否有剩余时间
		if true {
			hub.mu.RLock()
			hub.viewer = c.UserId
			hub.register <- c
			hub.mu.RUnlock()
			return nil

		} else {
			return nil
		}
	}
	if len(hub.clients) == 0 {
		hub.mu.RLock()
		hub.master = c.UserId
		hub.register <- c
		hub.mu.RUnlock()
		return nil
	}

	return nil
}

func (c *Client) WritePump() {
	defer func() {
		if r := recover(); r != nil {
			// TODO 记录日志

			err := c.Conn.Close()
			if err != nil {
				log.Printf("WritePump Error closing connection after panic: %v", err)
				return
			}
		}

	}()

	for message := range c.Send {
		err := c.Conn.WriteMessage(websocket.TextMessage, message)
		if err != nil {
			break
		}
	}
}

// Heartbeat 心跳检测
func (c *Client) Heartbeat() {

	ticker := time.NewTicker(2 * time.Second)
	defer func() {
		ticker.Stop()
		if r := recover(); r != nil {
			log.Printf("Heartbeat recover: %v", r)
			if c.Conn != nil {
				// TODO 记录日志
				err := c.Conn.Close()
				if err != nil {
					log.Printf("Heartbeat Error closing connection: %v", err)
					return
				}
			}
		}
	}()
	failTimes := 0
	for {
		select {
		case <-ticker.C:
			err := c.Conn.WriteControl(websocket.PingMessage, []byte{}, time.Now().Add(time.Second))

			if err != nil {
				log.Printf("心跳失败，err:%v", err)
				failTimes += 1

				if failTimes >= 3 {
					log.Printf("心跳重连最大次数，err:%v", err)
					if c.RoomId != "" {
						MyServer.Hubs[c.RoomId].unregister <- c
					}
					return
				}
			} else {
				// 如果成功则重置为0
				failTimes = 0
				//log.Printf("发送心跳成功,用户id：%v", c.UserId)
			}
		}
	}
}
```

```golang
package types

type WsReq struct {
	Event string `json:"event"` // 事件 sub、unsub、error、req、resp
}

// JoinReq 登录请求结构体
type JoinReq struct {
	Event string `json:"event"` // 事件 sub、unsub、error、req、resp
	Data  struct {
		UserId string `json:"user_id"` // 设备id
		RoomId string `json:"room_id"` // 区块
	} `json:"data"`
}

// OutReq 登录请求结构体
type OutReq struct {
	Event string `json:"event"` // 事件 sub、unsub、error、req、resp
	Data  struct {
		RoomId string `json:"room_id"` // 区块
	} `json:"data"`
}
type WsResp struct {
	Event string `json:"event"`
}

// JoinResp 加入房间广播
type JoinResp struct {
	Event string       `json:"event"`
	Data  JoinRespData `json:"data"`
}

type JoinRespData struct {
	Count  int    `json:"count"`
	Master string `json:"master"`
	Viewer string `json:"viewer"`
}

// SignallingReq 信令数据
type SignallingReq struct {
	Event string `json:"event"`
	Data  struct {
		RoomId string      `json:"room_id"` // 区块
		Value  interface{} `json:"value"`
	} `json:"data"`
}
```
